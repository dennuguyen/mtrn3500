    #include <cmath>

// calculator.h
namespace Calculator {
  class BasicCalculator {
    public:
      BasicCalculator(double left_operand, double right_operand); 
      ~BasicCalculator() = default;

      double add();
      double subtract();
      double divide();
      double multiply();
    protected:
      void check_for_divide_by_zero();
      double left_operand;
      double right_operand;
  };

  // challenge
  class ScientificCalculator : public Calculator::BasicCalculator {
    public:
      ScientificCalculator(double left_operand, double right_operand) : BasicCalculator(left_operand, right_operand) {}
      double sine();
      double cosine();
      double tangent();
      double inverse_sine();
      double inverse_cosine();
      double inverse_tangent();
  };
}

// calculator.cpp
Calculator::BasicCalculator::BasicCalculator(double left_operand, double right_operand) {
  this->left_operand = left_operand;
  this->right_operand = right_operand;
}

void Calculator::BasicCalculator::check_for_divide_by_zero() {
  if (this->right_operand == 0) {
    throw std::runtime_error("basic-calculator: Division by zero error");
  }
}

double Calculator::BasicCalculator::add() {
  return this->left_operand + this->right_operand;
}

double Calculator::BasicCalculator::subtract() {
  return this->left_operand - this->right_operand;
}

double Calculator::BasicCalculator::divide() {
  this->check_for_divide_by_zero();
  return this->left_operand / this->right_operand;
}

double Calculator::BasicCalculator::multiply() {
  return this->left_operand * this->right_operand;
}

double Calculator::ScientificCalculator::sine() {
  this->check_for_divide_by_zero();
  return sin(this->divide());
}

double Calculator::ScientificCalculator::cosine() {
  this->check_for_divide_by_zero();
  return cos(this->divide());
}

double Calculator::ScientificCalculator::tangent() {
  this->check_for_divide_by_zero();
  return tan(this->divide());
}

double Calculator::ScientificCalculator::inverse_sine() {
  this->check_for_divide_by_zero();
  return asin(this->divide());
}

double Calculator::ScientificCalculator::inverse_cosine() {
  this->check_for_divide_by_zero();
  return acos(this->divide());
}

double Calculator::ScientificCalculator::inverse_tangent() {
  this->check_for_divide_by_zero();
  return atan(this->divide());
}

    Method overloading

    Method overloading refers to the action of declaring and implementing two or more methods of the same name which accept different input parameters. This allows the user to implement a method in different ways which can then behave in different ways depending on the input parameters they are given. Consider the below case for the ComplexNumber class, which has methods for setting the value (real and imaginary components) of the complex number:

    namespace NewComplex {
  class ComplexNumber {
    public:
      // EXTENSION: overloaded constructors - what does each constructor do?
      ComplexNumber() : ComplexNumber(0, 0) {}
      ComplexNumber(double real) : ComplexNumber(real, 0) {}
      ComplexNumber(double real, double imag) : real(real), imag(imag) {}
      // use the default destructor as no resource allocation or clean-up is required
      ~ComplexNumber() = default;

      // operator overloads
      ComplexNumber operator+(const ComplexNumber& complex_number);
      ComplexNumber& operator+=(const ComplexNumber& complex_number);
      bool operator==(const ComplexNumber& complex_number);

      // overloaded `set_value` methods
      void set_value(double real);
      void set_value(double real, double imag);

      // getter functions
      double get_real() { return this->real; }
      double get_imag() { return this->imag; }

    private:
      double real;
      double imag;
  };
}

    Alright - there is a lot of new content here. Lets cover it one step at a time:

    Methods

    void NewComplex::ComplexNumber::set_value(double real) {
  this->set_value(real, 0);
}

void NewComplex::ComplexNumber::set_value(double real, double imag) {
  this->real = real;
  this->imag = imag;
}

    Here, the set_value method has been overloaded. Now, the method which returns a void can accept two combinations of parameters: one with a single double representing the new real component of the complex number, and one which accepts two double parameters representing the new real and imaginary components respectively. See how the first implementation calls the second implementation? This is an example of one use of overloaded methods when there are default values which can be used for the convenience and ease-of-use by the user. It also allows more complex details of the method to not be required to be provided unless the user requires them, for which they would then use the second implementation.

    // Why does this operator overload return a reference?
NewComplex::ComplexNumber& NewComplex::ComplexNumber::operator+=(const NewComplex::ComplexNumber& complex_number) {
  // why are we using '.' instead of '->'?
  this->real += complex_number.real;
  this->imag += complex_number.imag;
  // what are we actually returning?
  return *this;
}

NewComplex::ComplexNumber NewComplex::ComplexNumber::operator+(const ComplexNumber& complex_number) {
  // why are we making a copy of the current object?
  NewComplex::ComplexNumber result = *this;
  result += complex_number;
  // why are we returning the copy? Why are we returning anything?
  return result;
}

bool NewComplex::ComplexNumber::operator==(const NewComplex::ComplexNumber& complex_number) {
  // the below expression evaluates the equality condition and stores the boolean result 
  // in the respective variable
  bool real_match = this->real == complex_number.real;
  bool imag_match = this->imag == complex_number.imag;
  return real_match && imag_match;
}

    Here, we can observe that we are implementing methods in this class which begin with operator. These methods allow for the behaviour of operators (such as +, +=, /, and many more) to be overridden when referring to objects of the class they are implemented in. Note that these operators operate on the object itself.

    Copy Constructor

    namespace NewComplex {
  class ComplexNumber {
    public:
      // ...
      ComplexNumber(const ComplexNumber& to_copy_from);
      // ...
    private:
      double real;
      double imag;
  };
}

    Consider the case where you want to make a copy of a variable which you wish to modify instead of modifying the original parameter which was passed in. This is trivial when dealing with primitive data types (in-built data types, such as int and char) in C++, however more effort is required if this behaviour is required when implementing your own classes.

    This can be achieved using a special type of constructor called a copy constructor, which essentially takes a ‘frozen’ (thus const) copy of the variable at its current value and uses this to construct an (usually, however not always) identical copy of the object, which exists in an entirely distinct location in memory. Since this is still a constructor, it returns the instance of the object.

    NewComplex::ComplexNumber::ComplexNumber(const NewComplex::ComplexNumber& to_copy_from) {
  this->real = to_copy_from.real;
  this->imag = to_copy_from.imag;
}

    This is invoked similar to that of a constructor:

    NewComplex::ComplexNumber complex_number(1, 2);

NewComplex::ComplexNumber copy_constructed_complex_number(complex_number);

// NOTE: This is **NOT** the same as:
NewComplex::ComplexNumber assignment_operator_overload = complex_number;

    An example of its implementation is very similar to a standard constructor, except all the parameters must be inferred from the object being passed in as an argument. All member variables of the passed-in object (including protected and private) are accessible.

    Inheritance

    Inheritance is the term which refers to gaining the functionality of an existing class, which the new class is then based off. This is a core concept in an Object Oriented Language (C++ is an Object Oriented Language). When referring to inheritance, the terms parent refers to the class which the derived class (the child class) bases its initial structure off. Consider the below example where a vehicle class is derived upon by various child classes which derive from the vehicle parent class to gain the public functionality of the vehicle class (only public inheritance will be used for now):

    class Vehicle {
  public: 
    Vehicle();
    ~Vehicle();
  protected:
    // accessible from all child classes as if it were a private method / variable
    void set_amt_wheels(unsigned int);
  private:
    // NOT accessible from derived classes
    int secret_vehicle_variable;
};

class Car : public Vehicle {
  public: 
    Car();
    ~Car();
};

class Plane : public Vehicle {
  public: 
    Plane();
    ~Plane();
};

class Unicycle: public Vehicle {
  public: 
    Unicycle();
    ~Unicycle();
};
